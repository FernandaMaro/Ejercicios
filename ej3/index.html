const numeros = [1, 2, 3, 4];

function resultado (numeros) {
  let suma = 0;
  let product = 1;

(let i = 0; i < numeros.length; i++) {
  suma += numeros[i];
  product *= numeros[i];
}

console.log(`La suma es ${suma}.`);
console.log(`La multiplicacion (producto) es ${product}.`);
}
resultado(numeros);
